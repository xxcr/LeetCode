/**
 * @param {number[]} nums
 * @return {number}

var lengthOfLIS = function(nums) {
    let dp = [1]
    const n = nums.length

    for (let i = 1; i < n; i++) {
        dp[i] = 1
        for (let j = i - 1; j >= 0; j--) {
            if (nums[i] > nums[j]) {
                dp[i] = Math.max(dp[i], dp[j] + 1)
            }
        }
    }

    return Math.max(...dp)
}
